\newcommand{\includekanda}[4]{%
  \chapt{#1}%
  \foreach \n in {1,...,#4}{%
    % zero-pad to 3 digits
    \def\paddednum{\ifnum\n<10 00\n\else\ifnum\n<100 0\n\else\n\fi\fi}%
    \subimport{TeX/#2/}{#3-sarga-\paddednum.tex}%
  }%
}
